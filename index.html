Código de TensorFlow con Comentarios

 
# Importa TensorFlow, una biblioteca de machine learning.

            import tensorflow as tf

            
# Importa NumPy, que es útil para trabajar con arreglos y cálculos numéricos.

            import numpy as np

            
# Define los datos de entrenamiento. Los valores de temperatura en grados Celsius.

            celsius = np.array([-40, -10, 0, 8, 15, 22, 38], dtype=float)

            
# Define los valores correspondientes de temperatura en grados Fahrenheit.

            fahrenheit = np.array([-40, 14, 32, 46, 59, 72, 100], dtype=float)

            
# Crea la primera capa oculta con 3 neuronas y una forma de entrada de 1 .

            oculta1 = tf.keras.layers.Dense(units=3, input_shape=[1])

            
# Crea la segunda capa oculta, también con 3 neuronas.

            oculta2 = tf.keras.layers.Dense(units=3)

            
# Crea la capa de salida con una sola neurona, que generará el resultado de la predicción.

            salida = tf.keras.layers.Dense(units=1)

            
# Define el modelo secuencial con las capas creadas: oculta1, oculta2, y la capa de salida.

            modelo = tf.keras.Sequential([oculta1, oculta2, salida])

            
# Compila el modelo, especificando el optimizador y la función de pérdida.

            modelo.compile(optimizer=tf.keras.optimizers.Adam(0.1), loss='mean_squared_error')

            
# Mensaje en la consola para indicar el comienzo del entrenamiento.

            print("Comenzando entrenamiento...")

            
# Entrena el modelo con los datos de Celsius y Fahrenheit durante 1000 épocas. El parámetro 'verbose=False' desactiva la salida detallada del entrenamiento.

            historial = modelo.fit(celsius, fahrenheit, epochs=1000, verbose=False)

            
# Mensaje en la consola indicando que el modelo ha sido entrenado.

            print("Modelo entrenado!")

            
# Importa la biblioteca matplotlib para graficar los resultados de entrenamiento.

            import matplotlib.pyplot as plt

            
# Etiqueta para el eje X del gráfico, indicando el número de época.

            plt.xlabel("# Epoca")

            
# Etiqueta para el eje Y del gráfico, que muestra la magnitud de la pérdida.

            plt.ylabel("Magnitud de pérdida")

            
# Grafica la pérdida a lo largo de las épocas, usando los datos de historial del entrenamiento.

            plt.plot(historial.history["loss"])

            
# Mensaje en la consola para indicar que se hará una predicción con el modelo.

            print("Hagamos una predicción!")

            
# Usa el modelo entrenado para predecir el valor de Fahrenheit correspondiente a 100 grados Celsius.

            resultado = modelo.predict([100.0])

            
# Imprime el resultado de la predicción en la consola.

            print("El resultado es " + str(resultado) + " fahrenheit!")

            
# Mensaje en la consola para mostrar los pesos internos del modelo.

            print("Variables internas del modelo")

            
# Muestra los pesos de la primera capa oculta.

            print(oculta1.get_weights())

            
# Muestra los pesos de la segunda capa oculta.

            print(oculta2.get_weights())

            
# Muestra los pesos de la segunda capa oculta.

            print(salida.get_weights())
